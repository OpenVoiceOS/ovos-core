184 def _handle_transformers(self, message): 

185 """ 

186 Pipe utterance through transformer plugins to get more metadata. 

187 Utterances may be modified by any parser and context overwritten 

188 """ 

189 lang = get_message_lang(message) # per query lang or default Configuration lang 

190 original = utterances = message.data.get('utterances', []) 

191 message.context["lang"] = lang 

192 utterances, message.context = self.utterance_plugins.transform(utterances, message.context) 

193 if original != utterances: 

194 message.data["utterances"] = utterances 

195 LOG.debug(f"utterances transformed: {original} -> {utterances}") 

196 message.context = self.metadata_plugins.transform(message.context) 

197 return message

199 @staticmethod 

200 def disambiguate_lang(message): 

201 """ disambiguate language of the query via pre-defined context keys 

202 1 - stt_lang -> tagged in stt stage (STT used this lang to transcribe speech) 

203 2 - request_lang -> tagged in source message (wake word/request volunteered lang info) 

204 3 - detected_lang -> tagged by transformers (text classification, free form chat) 

205 4 - config lang (or from message.data) 

206 """ 

207 default_lang = get_message_lang(message) 

208 valid_langs = get_valid_languages() 

209 valid_langs = [standardize_lang_tag(l) for l in valid_langs] 

210 lang_keys = ["stt_lang", 

211 "request_lang", 

212 "detected_lang"] 

213 for k in lang_keys: 

214 if k in message.context: 

215 v = standardize_lang_tag(message.context[k]) 

216 if v in valid_langs: # TODO - use lang distance instead to choose best dialect 

217 if v != default_lang: 

218 LOG.info(f"replaced {default_lang} with {k}: {v}") 

219 return v 

220 else: 

221 LOG.warning(f"ignoring {k}, {v} is not in enabled languages: {valid_langs}") 

222 

223 return default_lang

225 def get_pipeline(self, skips=None, session=None) -> Tuple[str, Callable]: 

226 """ 

227 Constructs and returns the ordered list of intent matcher functions for the current session. 

228  

229 The pipeline sequence is determined by the session's configuration and may be filtered by 

230 an optional list of pipeline keys to skip. Each entry in the returned list is a tuple of 

231 the pipeline key and its corresponding matcher function, in the order they will be applied 

232 for intent matching. If a requested pipeline component is unavailable, it is skipped and a 

233 warning is logged. 

234  

235 Args: 

236 skips: Optional list of pipeline keys to exclude from the matcher sequence. 

237 session: Optional session object; if not provided, the current session is used. 

238  

239 Returns: 

240 A list of (pipeline_key, matcher_function) tuples representing the active intent 

241 matching pipeline for the session. 

242 """ 

243 session = session or SessionManager.get() 

244 

245 # Create matchers 

246 # TODO - from plugins 

247 padatious_matcher = None 

248 if self._padatious_service is None: 

249 needs_pada = any("padatious" in p for p in session.pipeline) 

250 if self._padacioso_service is not None: 

251 if needs_pada: 

252 LOG.warning("padatious is not available! using padacioso in it's place, " 

253 "intent matching will be extremely slow in comparison") 

254 padatious_matcher = self._padacioso_service 

255 elif needs_pada: 

256 LOG.warning("padatious is not available! only adapt (keyword based) intents will match!") 

257 else: 

258 padatious_matcher = self._padatious_service 

259 

260 matchers = { 

261 "converse": self._converse.converse_with_skills, 

262 "stop_high": self._stop.match_stop_high, 

263 "stop_medium": self._stop.match_stop_medium, 

264 "stop_low": self._stop.match_stop_low, 

265 "adapt_high": self._adapt_service.match_high, 

266 "common_qa": self._common_qa.match, 

267 "fallback_high": self._fallback.high_prio, 

268 "adapt_medium": self._adapt_service.match_medium, 

269 "fallback_medium": self._fallback.medium_prio, 

270 "adapt_low": self._adapt_service.match_low, 

271 "fallback_low": self._fallback.low_prio, 

272 "ovos-persona-pipeline-plugin-high": self._persona.match_high, 

273 "ovos-persona-pipeline-plugin-low": self._persona.match_low 

274 } 

275 if self._ollama is not None: 

276 matchers["ovos-ollama-intent-pipeline"] = self._ollama.match_low 

277 if self._m2v is not None: 

278 matchers["ovos-m2v-pipeline-high"] = self._m2v.match_high 

279 matchers["ovos-m2v-pipeline-medium"] = self._m2v.match_medium 

280 matchers["ovos-m2v-pipeline-low"] = self._m2v.match_low 

281 if self._padacioso_service is not None: 

282 matchers.update({ 

283 "padacioso_high": self._padacioso_service.match_high, 

284 "padacioso_medium": self._padacioso_service.match_medium, 

285 "padacioso_low": self._padacioso_service.match_low, 

286 

287 }) 

288 if padatious_matcher is not None: 

289 matchers.update({ 

290 "padatious_high": padatious_matcher.match_high, 

291 "padatious_medium": padatious_matcher.match_medium, 

292 "padatious_low": padatious_matcher.match_low, 

293 

294 }) 

295 if self._ocp is not None: 

296 matchers.update({ 

297 "ocp_high": self._ocp.match_high, 

298 "ocp_medium": self._ocp.match_medium, 

299 "ocp_fallback": self._ocp.match_fallback, 

300 "ocp_legacy": self._ocp.match_legacy}) 

301 skips = skips or [] 

302 pipeline = [k for k in session.pipeline if k not in skips] 

303 if any(k not in matchers for k in pipeline): 

304 LOG.warning(f"Requested some invalid pipeline components! " 

305 f"filtered {[k for k in pipeline if k not in matchers]}") 

306 pipeline = [k for k in pipeline if k in matchers] 

307 LOG.debug(f"Session pipeline: {pipeline}") 

308 return [(k, matchers[k]) for k in pipeline]

310 @staticmethod 

311 def _validate_session(message, lang): 

312 # get session 

313 lang = standardize_lang_tag(lang) 

314 sess = SessionManager.get(message) 

315 if sess.session_id == "default": 

316 updated = False 

317 # Default session, check if it needs to be (re)-created 

318 if sess.expired(): 

319 sess = SessionManager.reset_default_session() 

320 updated = True 

321 if lang != sess.lang: 

322 sess.lang = lang 

323 updated = True 

324 if updated: 

325 SessionManager.update(sess) 

326 SessionManager.sync(message) 

327 else: 

328 sess.lang = lang 

329 SessionManager.update(sess) 

330 sess.touch() 

331 return sess

333 def _handle_deactivate(self, message): 

334 """internal helper, track if a skill asked to be removed from active list during intent match 

335 in this case we want to avoid reactivating it again 

336 This only matters in PipelineMatchers, such as fallback and converse 

337 in those cases the activation is only done AFTER the match, not before unlike intents 

338 """ 

339 sess = SessionManager.get(message) 

340 skill_id = message.data.get("skill_id") 

341 self._deactivations[sess.session_id].append(skill_id)

343 def _emit_match_message(self, match: Union[IntentHandlerMatch, PipelineMatch], message: Message, lang: str): 

344 """ 

345 Emit a reply message for a matched intent, updating session and skill activation. 

346 

347 This method processes matched intents from either a pipeline matcher or an intent handler, 

348 creating a reply message with matched intent details and managing skill activation. 

349 

350 Args: 

351 match (Union[IntentHandlerMatch, PipelineMatch]): The matched intent object containing 

352 utterance and matching information. 

353 message (Message): The original messagebus message that triggered the intent match. 

354 lang (str): The language of the pipeline plugin match 

355 

356 Details: 

357 - Handles two types of matches: PipelineMatch and IntentHandlerMatch 

358 - Creates a reply message with matched intent data 

359 - Activates the corresponding skill if not previously deactivated 

360 - Updates session information 

361 - Emits the reply message on the messagebus 

362 

363 Side Effects: 

364 - Modifies session state 

365 - Emits a messagebus event 

366 - Can trigger skill activation events 

367 

368 Returns: 

369 None 

370 """ 

371 try: 

372 match = self.intent_plugins.transform(match) 

373 except Exception as e: 

374 LOG.error(f"Error in IntentTransformers: {e}") 

375 

376 reply = None 

377 sess = match.updated_session or SessionManager.get(message) 

378 sess.lang = lang # ensure it is updated 

379 

380 # utterance fully handled by pipeline matcher 

381 if isinstance(match, PipelineMatch): 

382 if match.handled: 

383 reply = message.reply("ovos.utterance.handled", {"skill_id": match.skill_id}) 

384 

385 # upload intent metrics if enabled 

386 create_daemon(self._upload_match_data, (match.utterance, 

387 match.skill_id, 

388 lang, 

389 match.match_data)) 

390 

391 # Launch skill if not handled by the match function 

392 elif isinstance(match, IntentHandlerMatch) and match.match_type: 

393 # keep all original message.data and update with intent match 

394 data = dict(message.data) 

395 data.update(match.match_data) 

396 reply = message.reply(match.match_type, data) 

397 

398 # upload intent metrics if enabled 

399 create_daemon(self._upload_match_data, (match.utterance, 

400 match.match_type, 

401 lang, 

402 match.match_data)) 

403 

404 if reply is not None: 

405 reply.data["utterance"] = match.utterance 

406 reply.data["lang"] = lang 

407 

408 # update active skill list 

409 if match.skill_id: 

410 # ensure skill_id is present in message.context 

411 reply.context["skill_id"] = match.skill_id 

412 

413 # NOTE: do not re-activate if the skill called self.deactivate 

414 # we could also skip activation if skill is already active, 

415 # but we still want to update the timestamp 

416 was_deactivated = match.skill_id in self._deactivations[sess.session_id] 

417 if not was_deactivated: 

418 sess.activate_skill(match.skill_id) 

419 # emit event for skills callback -> self.handle_activate 

420 self.bus.emit(reply.forward(f"{match.skill_id}.activate")) 

421 

422 # update Session if modified by pipeline 

423 reply.context["session"] = sess.serialize() 

424 

425 # finally emit reply message 

426 self.bus.emit(reply) 

427 

428 else: # upload intent metrics if enabled 

429 create_daemon(self._upload_match_data, (match.utterance, 

430 "complete_intent_failure", 

431 lang, 

432 match.match_data))

434 @staticmethod 

435 def _upload_match_data(utterance: str, intent: str, lang: str, match_data: dict): 

436 """if enabled upload the intent match data to a server, allowing users and developers 

437 to collect metrics/datasets to improve the pipeline plugins and skills. 

438 

439 There isn't a default server to upload things too, users needs to explicitly configure one 

440 

441 https://github.com/OpenVoiceOS/ovos-opendata-server 

442 """ 

443 config = Configuration().get("open_data", {}) 

444 endpoints: List[str] = config.get("intent_urls", []) # eg. "http://localhost:8000/intents" 

445 if not endpoints: 

446 return # user didn't configure any endpoints to upload metrics to 

447 if isinstance(endpoints, str): 

448 endpoints = [endpoints] 

449 headers = {"Content-Type": "application/x-www-form-urlencoded", 

450 "User-Agent": config.get("user_agent", "ovos-metrics")} 

451 data = { 

452 "utterance": utterance, 

453 "intent": intent, 

454 "lang": lang, 

455 "match_data": json.dumps(match_data, ensure_ascii=False) 

456 } 

457 for url in endpoints: 

458 try: 

459 # Add a timeout to prevent hanging 

460 response = requests.post(url, data=data, headers=headers, timeout=3) 

461 LOG.info(f"Uploaded intent metrics to '{url}' - Response: {response.status_code}") 

462 except Exception as e: 

463 LOG.warning(f"Failed to upload metrics: {e}")

465 def send_cancel_event(self, message): 

466 """ 

467 Emit events and play a sound when an utterance is canceled. 

468  

469 Logs the cancellation with the specific cancel word, plays a predefined cancel sound, 

470 and emits multiple events to signal the utterance cancellation. 

471  

472 Parameters: 

473 message (Message): The original message that triggered the cancellation. 

474  

475 Events Emitted: 

476 - 'mycroft.audio.play_sound': Plays a cancel sound from configuration 

477 - 'ovos.utterance.cancelled': Signals that the utterance was canceled 

478 - 'ovos.utterance.handled': Indicates the utterance processing is complete 

479  

480 Notes: 

481 - Uses the default cancel sound path 'snd/cancel.mp3' if not specified in configuration 

482 - Ensures events are sent as replies to the original message 

483 """ 

484 LOG.info("utterance canceled, cancel_word:" + message.context.get("cancel_word")) 

485 # play dedicated cancel sound 

486 sound = Configuration().get('sounds', {}).get('cancel', "snd/cancel.mp3") 

487 # NOTE: message.reply to ensure correct message destination 

488 self.bus.emit(message.reply('mycroft.audio.play_sound', {"uri": sound})) 

489 self.bus.emit(message.reply("ovos.utterance.cancelled")) 

490 self.bus.emit(message.reply("ovos.utterance.handled"))

492 def handle_utterance(self, message: Message): 

493 """Main entrypoint for handling user utterances 

494 

495 Monitor the messagebus for 'recognizer_loop:utterance', typically 

496 generated by a spoken interaction but potentially also from a CLI 

497 or other method of injecting a 'user utterance' into the system. 

498 

499 Utterances then work through this sequence to be handled: 

500 1) UtteranceTransformers can modify the utterance and metadata in message.context 

501 2) MetadataTransformers can modify the metadata in message.context 

502 3) Language is extracted from message 

503 4) Active skills attempt to handle using converse() 

504 5) Padatious high match intents (conf > 0.95) 

505 6) Adapt intent handlers 

506 7) CommonQuery Skills 

507 8) High Priority Fallbacks 

508 9) Padatious near match intents (conf > 0.8) 

509 10) General Fallbacks 

510 11) Padatious loose match intents (conf > 0.5) 

511 12) Catch all fallbacks including Unknown intent handler 

512 

513 If all these fail the complete_intent_failure message will be sent 

514 and a generic error sound played. 

515 

516 Args: 

517 message (Message): The messagebus data 

518 """ 

519 # Get utterance utterance_plugins additional context 

520 message = self._handle_transformers(message) 

521 

522 if message.context.get("canceled"): 

523 self.send_cancel_event(message) 

524 return 

525 

526 # tag language of this utterance 

527 lang = self.disambiguate_lang(message) 

528 

529 utterances = message.data.get('utterances', []) 

530 LOG.info(f"Parsing utterance: {utterances}") 

531 

532 stopwatch = Stopwatch() 

533 

534 # get session 

535 sess = self._validate_session(message, lang) 

536 message.context["session"] = sess.serialize() 

537 

538 # match 

539 match = None 

540 with stopwatch: 

541 self._deactivations[sess.session_id] = [] 

542 # Loop through the matching functions until a match is found. 

543 for pipeline, match_func in self.get_pipeline(session=sess): 

544 langs = [lang] 

545 if self.config.get("multilingual_matching"): 

546 # if multilingual matching is enabled, attempt to match all user languages if main fails 

547 langs += [l for l in get_valid_languages() if l != lang] 

548 for intent_lang in langs: 

549 match = match_func(utterances, intent_lang, message) 

550 if match: 

551 LOG.info(f"{pipeline} match ({intent_lang}): {match}") 

552 if match.skill_id and match.skill_id in sess.blacklisted_skills: 

553 LOG.debug( 

554 f"ignoring match, skill_id '{match.skill_id}' blacklisted by Session '{sess.session_id}'") 

555 continue 

556 if isinstance(match, IntentHandlerMatch) and match.match_type in sess.blacklisted_intents: 

557 LOG.debug( 

558 f"ignoring match, intent '{match.match_type}' blacklisted by Session '{sess.session_id}'") 

559 continue 

560 try: 

561 self._emit_match_message(match, message, intent_lang) 

562 break 

563 except: 

564 LOG.exception(f"{match_func} returned an invalid match") 

565 else: 

566 LOG.debug(f"no match from {match_func}") 

567 continue 

568 break 

569 else: 

570 # Nothing was able to handle the intent 

571 # Ask politely for forgiveness for failing in this vital task 

572 self.send_complete_intent_failure(message) 

573 

574 LOG.debug(f"intent matching took: {stopwatch.time}") 

575 

576 # sync any changes made to the default session, eg by ConverseService 

577 if sess.session_id == "default": 

578 SessionManager.sync(message) 

579 elif sess.session_id in self._deactivations: 

580 self._deactivations.pop(sess.session_id) 

581 return match, message.context, stopwatch

583 def send_complete_intent_failure(self, message): 

584 """Send a message that no skill could handle the utterance. 

585 

586 Args: 

587 message (Message): original message to forward from 

588 """ 

589 sound = Configuration().get('sounds', {}).get('error', "snd/error.mp3") 

590 # NOTE: message.reply to ensure correct message destination 

591 self.bus.emit(message.reply('mycroft.audio.play_sound', {"uri": sound})) 

592 self.bus.emit(message.reply('complete_intent_failure')) 

593 self.bus.emit(message.reply("ovos.utterance.handled"))

595 @staticmethod 

596 def handle_add_context(message: Message): 

597 """Add context 

598 

599 Args: 

600 message: data contains the 'context' item to add 

601 optionally can include 'word' to be injected as 

602 an alias for the context item. 

603 """ 

604 entity = {'confidence': 1.0} 

605 context = message.data.get('context') 

606 word = message.data.get('word') or '' 

607 origin = message.data.get('origin') or '' 

608 # if not a string type try creating a string from it 

609 if not isinstance(word, str): 

610 word = str(word) 

611 entity['data'] = [(word, context)] 

612 entity['match'] = word 

613 entity['key'] = word 

614 entity['origin'] = origin 

615 sess = SessionManager.get(message) 

616 sess.context.inject_context(entity)

636 def handle_get_intent(self, message): 

637 """Get intent from either adapt or padatious. 

638 

639 Args: 

640 message (Message): message containing utterance 

641 """ 

642 utterance = message.data["utterance"] 

643 lang = get_message_lang(message) 

644 sess = SessionManager.get(message) 

645 match = None 

646 # Loop through the matching functions until a match is found. 

647 for pipeline, match_func in self.get_pipeline(skips=["converse", 

648 "common_qa", 

649 "fallback_high", 

650 "fallback_medium", 

651 "fallback_low"], 

652 session=sess): 

653 s = time.monotonic() 

654 match = match_func([utterance], lang, message) 

655 LOG.debug(f"matching '{pipeline}' took: {time.monotonic() - s} seconds") 

656 if match: 

657 if match.match_type: 

658 intent_data = dict(match.match_data) 

659 intent_data["intent_name"] = match.match_type 

660 intent_data["intent_service"] = pipeline 

661 intent_data["skill_id"] = match.skill_id 

662 intent_data["handler"] = match_func.__name__ 

663 LOG.debug(f"final intent match: {intent_data}") 

664 m = message.reply("intent.service.intent.reply", 

665 {"intent": intent_data, "utterance": utterance}) 

666 self.bus.emit(m) 

667 return 

668 LOG.error(f"bad pipeline match! {match}") 

669 # signal intent failure 

670 self.bus.emit(message.reply("intent.service.intent.reply", 

671 {"intent": None, "utterance": utterance}))

697 self.bus.remove('add_context', self.handle_add_context) 

698 self.bus.remove('remove_context', self.handle_remove_context) 

699 self.bus.remove('clear_context', self.handle_clear_context) 

700 self.bus.remove('mycroft.skills.loaded', self.update_skill_name_dict) 

701 self.bus.remove('intent.service.intent.get', self.handle_get_intent) 

702 self.bus.remove('intent.service.skills.get', self.handle_get_skills)
